# Uncontrolled vs Controlled Tabs

By default, the `<Tabs />` component controlls the current tab internally.

But you can take control of the this using the `currentTab` and `onChange` props.
This is useful for integrating the Tab component into things like react-router.

```tsx
import { useMemo } from "react";
import { useLocation, useHistory } from "react-router-dom";
import { Tabs, Tab } from "atomic-elements";

const ROUTES = { "/preview": "tab1", "/edit": "tab2" };

function ControlledTabs() {
  const history = useHistory();
  const { pathname } = useLocation();

  const currentTab = ROUTES[pathname];

  const onChange = (tab: string) => {
    const route = Object.entries(ROUTES).find(
      ([route, tabName]) => tab === tabName
    )[0];
    history.push(route);
  };

  return (
    <Tabs
      tabs={{ tab1: "Tab One", tab2: "Tab Two" }}
      currentTab={currentTab}
      onChange={onChange}
    >
      <Tab name="tab1">
        <div> I'm the first tab</div>
      </Tab>

      <Tab name="tab2">
        <div> I'm the second tab</div>
      </Tab>
    </Tabs>
  );
}
```
